Transit AI Chat - Widget Flotante

Haz clic en el boton azul de la esquina inferior derecha para abrir el chat. El widget se comunica con la pagina padre via postMessage.

<!-- 1. Copiar este snippet antes de </body> -->

<!-- Boton flotante -->
<button id="chat-widget-btn" onclick="toggleChat()"
  style="position:fixed;bottom:24px;right:24px;z-index:9999;
  width:56px;height:56px;border-radius:50%;border:none;
  background:#3b82f6;color:#fff;cursor:pointer;font-size:24px;
  box-shadow:0 4px 16px rgba(59,130,246,0.4)">💬</button>

<!-- Panel del chat -->
<div id="chat-widget-panel"
  style="position:fixed;bottom:92px;right:24px;z-index:9998;
  width:400px;height:600px;border-radius:16px;overflow:hidden;
  box-shadow:0 8px 40px rgba(0,0,0,0.2);display:none">
  <iframe id="transit-chat"
    src="https://validacion.trafficsoft.co/chat/"
    style="width:100%;height:100%;border:none"
    allow="clipboard-write"></iframe>
</div>

<script>
  var chatOpen = false;
  function toggleChat() {
    chatOpen = !chatOpen;
    var panel = document.getElementById('chat-widget-panel');
    panel.style.display = chatOpen ? 'block' : 'none';
  }

  // Enviar contexto cuando el chat este listo
  window.addEventListener('message', function(e) {
    if (e.data && e.data.type === 'chat-ready') {
      document.getElementById('transit-chat')
        .contentWindow.postMessage({
          type: 'set-user',
          userId: 'nombre.usuario',  // Reemplazar
          city: 'bogota'             // Opcional
        }, 'https://validacion.trafficsoft.co');
    }
  });
</script>
