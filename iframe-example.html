Transit AI Chat - Ejemplo de Inyeccion iframe

El chat se comunica con la pagina padre via postMessage para recibir el nombre del usuario.

<!-- 1. Agregar el iframe en tu pagina -->
<iframe
  id="transit-chat"
  src="https://validacion.trafficsoft.co/chat/"
  style="width: 100%; height: 700px; border: none; border-radius: 12px;"
  allow="clipboard-write"
></iframe>

<script>
  // 2. Esperar a que el chat este listo
  window.addEventListener('message', function(e) {
    if (e.data?.type === 'chat-ready') {
      // 3. Enviar el userId al chat
      var iframe = document.getElementById('transit-chat');
      iframe.contentWindow.postMessage({
        type: 'set-user',
        userId: 'nombre.usuario'  // <-- Reemplazar con el usuario real
      }, 'https://validacion.trafficsoft.co');
    }
  });
</script>
